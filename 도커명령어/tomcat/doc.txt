# 이미지 mysql oracle 

docker run -d -e MYSQL_ROOT_PASSWORD=1234 --name mysqlcontainer mysql

# 이미지 다운로드
docker pull mysql

# 컨테이너 생성
docker create --name mysqlcontainer -e MYSQL_ROOT_PASSWORD=1234 -p 3306:3306 mysql
docker run -d --name mysqlcontainer -e MYSQL_ROOT_PASSWORD=1234 -p 3306:3306 -v d:/mysql:/var/lib/mysql mysql

docker network create --gateway 172.18.0.1 --subnet 172.18.0.0/21 mybridge
docker run --name mysql -d --network bridge -ip 172.17.0.10 -p 3306:3306 -e MYSQL_ROOT_PASSWORD=1234 -e MYSQL_CHARSET=utf8mb4 -e MYSQL_COLLATION=utf8mb4_unicode_ci mysql

docker run -d --name mysql-container -e MYSQL_ROOT_PASSWORD=examplepassword -e MYSQL_DATABASE=exampledb -e MYSQL_USER=exampleuser -e MYSQL_PASSWORD=examplepassword -e MYSQL_CHARSET=utf8mb4 -e MYSQL_COLLATION=utf8mb4_unicode_ci -p 3306:3306 mysql:latest

# 도커 모든 이미지 보기

# 도커 이미지 삭제 명령어

# 도커 컨테이너 삭제 명령어
docker rm mysqlcontainer

# 도커 컨테이너 목록 보기
docker ps
